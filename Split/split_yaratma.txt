import random

f=open("odev.txt","w")


urunler=["sut", "su", "soda","ekmek","peynir"]
subeler=["universite", "merkez", "hastane"]

for i in range(1000):
    f.write(random.choice(subeler)+" =>")
    for k in range(4):
        f.write(random.choice(urunler)+":"+str(random.randint(1,10))+" ")
    f.write("\n")

f.close()
